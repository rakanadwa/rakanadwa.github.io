  <script>

    // Initial zoom values

    const zoomStates = {};

   

    // Initialize all zoom levels

    document.addEventListener('DOMContentLoaded', function() {

      const zoomImages = document.querySelectorAll('.zoom-image');

      zoomImages.forEach(img => {

        const id = img.id;

        zoomStates[id] = {

          scale: 1,

          offsetX: 0,

          offsetY: 0,

          isDragging: false,

          lastX: 0,

          lastY: 0

        };

       

        // Add mouse events for dragging

        img.addEventListener('mousedown', function(e) {

          const state = zoomStates[id];

          state.isDragging = true;

          state.lastX = e.clientX;

          state.lastY = e.clientY;

        });

       

        img.addEventListener('mousemove', function(e) {

          const state = zoomStates[id];

          if (state.isDragging) {

            const deltaX = e.clientX - state.lastX;

            const deltaY = e.clientY - state.lastY;

           

            state.offsetX += deltaX / state.scale;

            state.offsetY += deltaY / state.scale;

           

            state.lastX = e.clientX;

            state.lastY = e.clientY;

           

            updateZoomImage(id);

          }

        });

       

        img.addEventListener('mouseup', function() {

          zoomStates[id].isDragging = false;

        });

       

        img.addEventListener('mouseleave', function() {

          zoomStates[id].isDragging = false;

        });

      });

    });

   

    // Function to change zoom level

    function changeZoom(imageId, delta) {

      const state = zoomStates[imageId];

      const newScale = Math.max(1, Math.min(10, state.scale + delta));

     

      // Update scale

      state.scale = newScale;

     

      // Update zoom level display

      const zoomLevelEl = document.getElementById(`${imageId}-level`);

      if (zoomLevelEl) {

        zoomLevelEl.textContent = `${Math.round(newScale * 100)}%`;

      }

     

      updateZoomImage(imageId);

    }

   

    // Function to update the zoomed image

    function updateZoomImage(imageId) {

      const img = document.getElementById(imageId);

      const state = zoomStates[imageId];

     

      if (img) {

        img.style.transform = `scale(${state.scale}) translate(${state.offsetX}px, ${state.offsetY}px)`;

        img.style.transformOrigin = 'center';

      }

    }
